/* collections.js
 * Server-side collections, accessible by both client and server
 *
 * Underlying data structures:
 * documents of the format {key:data type, ...}
 */

/* Users: {_id: string, 
 *         username: string, 
 *         groups: [{groupid: string}], 
 *         current: string}
 */
Users = new Mongo.Collection('Users');

/* Groups: {_id: string, 
 * groupname: string, 
 * groupdescription: string, 
 * employers: [{userid: string}], 
 * users: [{userid: string}]} 
 */
Groups = new Mongo.Collection('Groups');

/* we will have to split start and end time via ":"
 * Shifts: {_id: string, 
 * [date: string | weekday: int],  
 * start: string, 
 * end: string, 
 * users: [{userid: string}],
 * swaps: [{swapid: string, swapin: string, swapout: string, date: string}], 
 * groupid: string}
 */
Shifts = new Mongo.Collection('Shifts');

/* Notifications : {_id: string,
 * groupid: string,
 * sender: string,
 * shiftid: string,
 * date: string,
 * startin: string,
 * endin: string,
 * startout: string,
 * endout: string,
 * type: string,
 * swapid: string,
 * swapdate: string,
 * seen: [string],
 * ok: [string]}
 */
Notifications = new Mongo.Collection('Notifications');
